/*
 * MSP430F5529 Pocket Kit temperature logger
 *
 * Sensors:
 *   - MSP430F5529 internal temperature sensor, ADC12 channel A10.
 *   - External NTC thermistor on P6.5/A5 by default.
 *   - TMP421 on UCB0 I2C: P3.1=SCL, P3.0=SDA, address auto-detected.
 *
 * Outputs:
 *   - UART: P4.4=TX, P4.5=RX, 9600 8N1.
 *   - 2.13 in E-paper module through bit-banged 4-wire SPI.
 *   - Passive buzzer on P3.6, 2 kHz square-wave alarm.
 *
 * UART commands:
 *   h: print history and show the first history page on E-paper.
 *   n: show next history page on E-paper.
 *   c: collect one sample immediately.
 *   + / -: change alarm threshold by 1.0 C.
 *   e: erase history.
 *   i: scan I2C bus and print TMP421 ID.
 *   p/o/t: SSD1673 E-paper diagnostics.
 *   d/f/g/m/r: alternate SPD2701 diagnostics, select alternate driver, resume.
 *   u/s/x: SSD1673 RAM-bypass cycle, EPD pin register dump, pin drive test.
 *   ?: print help.
 */

static void delay_ms(uint16_t ms)
{
    while (ms--) {
        __delay_cycles(SMCLK_HZ / 1000u);
    }
}

static void clock_init(void)
{
    UCSCTL3 = SELREF_2;
    UCSCTL4 = SELA_2 | SELS_4 | SELM_4;
    __bis_SR_register(SCG0);
    UCSCTL0 = 0;
    UCSCTL1 = DCORSEL_2;
    UCSCTL2 = FLLD_1 | 31;
    __bic_SR_register(SCG0);
    delay_ms(250);
}

static void gpio_init(void)
{
    P1DIR |= BIT0;
    P1OUT &= ~BIT0;

    P8DIR |= BIT1;
    P8OUT &= ~BIT1;

    BUZZER_DIR |= BUZZER_BIT;
    BUZZER_OUT &= ~BUZZER_BIT;

    NTC_PORT_SEL |= NTC_PORT_BIT;
}

static void uart_init(void)
{
    P4SEL |= BIT4 | BIT5;
    UCA1CTL1 |= UCSWRST;
    UCA1CTL1 |= UCSSEL_2;
    UCA1BR0 = 6;
    UCA1BR1 = 0;
    UCA1MCTL = UCBRF_13 | UCOS16;
    UCA1CTL1 &= ~UCSWRST;
    UCA1IE |= UCRXIE;
}

static void uart_putc(char c)
{
    while (!(UCA1IFG & UCTXIFG)) {
        ;
    }
    UCA1TXBUF = (uint8_t)c;
}

static void uart_puts(const char *s)
{
    while (*s) {
        uart_putc(*s++);
    }
}

static void uart_print_u16(uint16_t v)
{
    char buf[6];
    uint8_t i;

    if (v == 0) {
        uart_putc('0');
        return;
    }

    i = 0;
    while (v > 0 && i < sizeof(buf)) {
        buf[i++] = (char)('0' + (v % 10));
        v = (uint16_t)(v / 10);
    }
    while (i > 0) {
        uart_putc(buf[--i]);
    }
}

static void uart_print_hex_u8(uint8_t v)
{
    const char hex[] = "0123456789ABCDEF";

    uart_puts("0x");
    uart_putc(hex[(v >> 4) & 0x0F]);
    uart_putc(hex[v & 0x0F]);
}

static void uart_print_i16_t10(int16_t t10)
{
    uint16_t a;

    if (t10 == INVALID_T10) {
        uart_puts("--.-");
        return;
    }

    if (t10 < 0) {
        uart_putc('-');
        a = (uint16_t)(-t10);
    } else {
        a = (uint16_t)t10;
    }

    uart_print_u16((uint16_t)(a / 10));
    uart_putc('.');
    uart_putc((char)('0' + (a % 10)));
}

static char *append_str(char *p, const char *s)
{
    while (*s) {
        *p++ = *s++;
    }
    *p = 0;
    return p;
}

static char *append_u16(char *p, uint16_t v)
{
    char buf[6];
    uint8_t i;

    if (v == 0) {
        *p++ = '0';
        *p = 0;
        return p;
    }

    i = 0;
    while (v > 0 && i < sizeof(buf)) {
        buf[i++] = (char)('0' + (v % 10));
        v = (uint16_t)(v / 10);
    }
    while (i > 0) {
        *p++ = buf[--i];
    }
    *p = 0;
    return p;
}

static char *append_t10(char *p, int16_t t10)
{
    uint16_t a;

    if (t10 == INVALID_T10) {
        return append_str(p, "--.-");
    }

    if (t10 < 0) {
        *p++ = '-';
        a = (uint16_t)(-t10);
    } else {
        a = (uint16_t)t10;
    }

    p = append_u16(p, (uint16_t)(a / 10));
    *p++ = '.';
    *p++ = (char)('0' + (a % 10));
    *p = 0;
    return p;
}

static uint8_t temp_is_valid(int16_t t10)
{
    return (uint8_t)(t10 != INVALID_T10);
}

static uint8_t sample_over_threshold(const TempSample *s)
{
    if (temp_is_valid(s->die_t10) && s->die_t10 >= g_threshold_t10) {
        return 1;
    }
    if (temp_is_valid(s->ntc_t10) && s->ntc_t10 >= g_threshold_t10) {
        return 1;
    }
    if (temp_is_valid(s->tmp_local_t10) && s->tmp_local_t10 >= g_threshold_t10) {
        return 1;
    }
    if (temp_is_valid(s->tmp_remote_t10) && s->tmp_remote_t10 >= g_threshold_t10) {
        return 1;
    }
    return 0;
}

static void uart_print_sample(const TempSample *s)
{
    uart_puts("Now  DIE=");
    uart_print_i16_t10(s->die_t10);
    uart_puts("C NTC=");
    uart_print_i16_t10(s->ntc_t10);
    uart_puts("C TMP=");
    uart_print_i16_t10(s->tmp_local_t10);
    uart_puts("C REM=");
    uart_print_i16_t10(s->tmp_remote_t10);
    uart_puts("C ALM=");
    uart_print_i16_t10(g_threshold_t10);
    uart_puts("C\r\n");
}

static void uart_print_help(void)
{
    uart_puts("\r\nCommands: h=history, n=next page, c=sample, +=alarm up, -=alarm down, e=erase, i=i2c, p/o/t=ssd epd, u=bypass, s=pins, x=pin test, d/f/g=alt epd, m=use alt, r=resume, ?=help\r\n");
}

static void adc_power_reference(uint8_t on)
{
    if (on) {
        REFCTL0 = REFMSTR | REFVSEL_0 | REFON;
        delay_ms(2);
    } else {
        REFCTL0 = REFMSTR;
    }
}

static uint16_t adc_read_once(uint16_t inch, uint16_t sref, uint8_t use_internal_ref)
{
    uint16_t value;
    uint16_t timeout;

    if (use_internal_ref) {
        adc_power_reference(1);
    }

    ADC12CTL0 &= ~ADC12ENC;
    ADC12CTL0 = ADC12SHT0_15 | ADC12ON;
    ADC12CTL1 = ADC12SHP;
    ADC12MCTL0 = sref | inch;
    ADC12CTL0 |= ADC12ENC | ADC12SC;

    timeout = 60000u;
    while ((ADC12CTL1 & ADC12BUSY) && timeout > 0) {
        timeout--;
    }

    value = ADC12MEM0;
    ADC12CTL0 &= ~ADC12ENC;
    ADC12CTL0 &= ~ADC12ON;

    if (use_internal_ref) {
        adc_power_reference(0);
    }

    return value;
}

static uint16_t adc_read_avg(uint16_t inch, uint16_t sref, uint8_t use_internal_ref)
{
    uint8_t i;
    uint32_t sum;

    sum = 0;
    for (i = 0; i < 4; i++) {
        sum += adc_read_once(inch, sref, use_internal_ref);
    }
    return (uint16_t)(sum / 4u);
}

static int16_t die_adc_to_t10(uint16_t adc)
{
    uint16_t cal30;
    uint16_t cal85;
    long t10;
    long mv;

    cal30 = CAL_ADC_15T30;
    cal85 = CAL_ADC_15T85;

    if (cal30 != 0xFFFFu && cal85 != 0xFFFFu && cal85 > cal30) {
        t10 = ((long)adc - (long)cal30) * 550L;
        t10 = t10 / ((long)cal85 - (long)cal30) + 300L;
        return (int16_t)t10;
    }

    mv = ((long)adc * 1500L) / 4095L;
    t10 = ((mv - 986L) * 1000L) / 355L;
    return (int16_t)t10;
}

static int16_t ntc_adc_to_t10(uint16_t adc)
{
    uint8_t i;
    uint16_t a0;
    uint16_t a1;
    int16_t t0;
    int16_t t1;
    long t;
    uint16_t scaled_adc;
    const uint8_t n = (uint8_t)(sizeof(ntc_table) / sizeof(ntc_table[0]));

#if NTC_PULLUP_TO_VCC
    scaled_adc = adc;
#else
    scaled_adc = (uint16_t)(4095u - adc);
#endif

    if (scaled_adc >= ntc_table[0].adc) {
        return ntc_table[0].t10;
    }
    if (scaled_adc <= ntc_table[n - 1u].adc) {
        return ntc_table[n - 1u].t10;
    }

    for (i = 0; i < (uint8_t)(n - 1u); i++) {
        a0 = ntc_table[i].adc;
        a1 = ntc_table[i + 1u].adc;
        if (scaled_adc <= a0 && scaled_adc >= a1) {
            t0 = ntc_table[i].t10;
            t1 = ntc_table[i + 1u].t10;
            t = (long)t0 + ((long)scaled_adc - (long)a0) * ((long)t1 - (long)t0) /
                ((long)a1 - (long)a0);
            return (int16_t)t;
        }
    }

    return INVALID_T10;
}

static void i2c_init(void)
{
    P3SEL |= BIT0 | BIT1;
    UCB0CTL1 |= UCSWRST;
    UCB0CTL0 = UCMST | UCMODE_3 | UCSYNC;
    UCB0CTL1 = UCSWRST | UCSSEL_2;
    UCB0BR0 = 10;
    UCB0BR1 = 0;
    UCB0CTL1 &= ~UCSWRST;
}

static uint8_t i2c_wait_bus_free(void)
{
    uint16_t timeout;

    timeout = 60000u;
    while ((UCB0STAT & UCBBUSY) && timeout > 0) {
        timeout--;
    }
    return (uint8_t)(timeout > 0);
}

static uint8_t i2c_check_nack(void)
{
    if (UCB0IFG & UCNACKIFG) {
        UCB0IFG &= ~UCNACKIFG;
        UCB0CTL1 |= UCTXSTP;
        return 1;
    }
    return 0;
}

static uint8_t i2c_wait_txifg(void)
{
    uint16_t timeout;

    timeout = 60000u;
    while (!(UCB0IFG & UCTXIFG) && timeout > 0) {
        if (i2c_check_nack()) {
            return 0;
        }
        timeout--;
    }
    return (uint8_t)(timeout > 0);
}

static uint8_t i2c_wait_rxifg(void)
{
    uint16_t timeout;

    timeout = 60000u;
    while (!(UCB0IFG & UCRXIFG) && timeout > 0) {
        if (i2c_check_nack()) {
            return 0;
        }
        timeout--;
    }
    return (uint8_t)(timeout > 0);
}

static uint8_t i2c_wait_stop(void)
{
    uint16_t timeout;

    timeout = 60000u;
    while ((UCB0CTL1 & UCTXSTP) && timeout > 0) {
        timeout--;
    }
    return (uint8_t)(timeout > 0);
}

static uint8_t i2c_probe(uint8_t addr)
{
    uint16_t timeout;
    uint8_t ack;

    if (!i2c_wait_bus_free()) {
        return 0;
    }

    UCB0I2CSA = addr;
    UCB0CTL1 |= UCTR | UCTXSTT;

    timeout = 60000u;
    ack = 0;
    while (timeout > 0) {
        if (UCB0IFG & UCNACKIFG) {
            UCB0IFG &= ~UCNACKIFG;
            ack = 0;
            break;
        }
        if (UCB0IFG & UCTXIFG) {
            ack = 1;
            break;
        }
        timeout--;
    }

    UCB0CTL1 |= UCTXSTP;
    (void)i2c_wait_stop();
    return ack;
}

static uint8_t i2c_write_reg(uint8_t addr, uint8_t reg, uint8_t value)
{
    if (!i2c_wait_bus_free()) {
        return 0;
    }

    UCB0I2CSA = addr;
    UCB0CTL1 |= UCTR | UCTXSTT;
    if (!i2c_wait_txifg()) {
        return 0;
    }
    UCB0TXBUF = reg;
    if (!i2c_wait_txifg()) {
        return 0;
    }
    UCB0TXBUF = value;
    if (!i2c_wait_txifg()) {
        return 0;
    }
    UCB0CTL1 |= UCTXSTP;
    return i2c_wait_stop();
}

static uint8_t i2c_read_reg(uint8_t addr, uint8_t reg, uint8_t *value)
{
    uint16_t timeout;

    if (!i2c_wait_bus_free()) {
        return 0;
    }

    UCB0I2CSA = addr;
    UCB0CTL1 |= UCTR | UCTXSTT;
    if (!i2c_wait_txifg()) {
        return 0;
    }
    UCB0TXBUF = reg;
    if (!i2c_wait_txifg()) {
        return 0;
    }

    UCB0CTL1 &= ~UCTR;
    UCB0CTL1 |= UCTXSTT;

    timeout = 60000u;
    while ((UCB0CTL1 & UCTXSTT) && timeout > 0) {
        if (i2c_check_nack()) {
            return 0;
        }
        timeout--;
    }
    if (timeout == 0) {
        UCB0CTL1 |= UCTXSTP;
        return 0;
    }

    UCB0CTL1 |= UCTXSTP;
    if (!i2c_wait_rxifg()) {
        return 0;
    }
    *value = UCB0RXBUF;
    return i2c_wait_stop();
}

static uint8_t tmp421_is_device_id(uint8_t id)
{
    return (uint8_t)(id == 0x21u || id == 0x22u || id == 0x23u ||
                     id == 0x41u || id == 0x42u);
}

static uint8_t tmp421_detect(void)
{
    const uint8_t candidates[] = {0x2Au, 0x4Cu, 0x4Du, 0x4Eu, 0x4Fu};
    uint8_t i;
    uint8_t manf;
    uint8_t dev;
    uint8_t local;

    for (i = 0; i < (uint8_t)sizeof(candidates); i++) {
        if (!i2c_probe(candidates[i])) {
            continue;
        }
        if (i2c_read_reg(candidates[i], TMP421_REG_MANUFACTURER_ID, &manf) &&
            i2c_read_reg(candidates[i], TMP421_REG_DEVICE_ID, &dev) &&
            manf == TMP421_MANUFACTURER_ID && tmp421_is_device_id(dev)) {
            g_tmp421_addr = candidates[i];
            return 1;
        }
        if (candidates[i] == 0x2Au &&
            i2c_read_reg(candidates[i], TMP421_REG_LOCAL_MSB, &local)) {
            g_tmp421_addr = candidates[i];
            return 1;
        }
        if (candidates[i] != 0x2Au) {
            continue;
        }
    }

    g_tmp421_addr = 0;
    return 0;
}

static void i2c_print_scan(void)
{
    uint8_t addr;
    uint8_t manf;
    uint8_t dev;

    uart_puts("I2C:");
    for (addr = 0x08u; addr < 0x78u; addr++) {
        if (i2c_probe(addr)) {
            uart_putc(' ');
            uart_print_hex_u8(addr);
        }
    }
    uart_puts("\r\n");

    if (g_tmp421_addr) {
        uart_puts("TMP421 addr=");
        uart_print_hex_u8(g_tmp421_addr);
        if (i2c_read_reg(g_tmp421_addr, TMP421_REG_MANUFACTURER_ID, &manf) &&
            i2c_read_reg(g_tmp421_addr, TMP421_REG_DEVICE_ID, &dev)) {
            uart_puts(" manf=");
            uart_print_hex_u8(manf);
            uart_puts(" dev=");
            uart_print_hex_u8(dev);
        }
        uart_puts("\r\n");
    } else {
        uart_puts("TMP421 not detected\r\n");
    }

    uart_puts("NTC P6.5/A5 raw=");
    uart_print_u16(g_last_ntc_adc);
    uart_puts("\r\n");
}

static void tmp421_init(void)
{
    if (!tmp421_detect()) {
        return;
    }

    (void)i2c_write_reg(g_tmp421_addr, TMP421_REG_CONFIG1_WRITE, TMP421_CONFIG_EXTENDED);
    (void)i2c_write_reg(g_tmp421_addr, TMP421_REG_CHANNEL_WRITE, TMP421_CHANNEL_LOCAL_REMOTE);
    (void)i2c_write_reg(g_tmp421_addr, TMP421_REG_CONV_RATE_WRITE, 0x07);
}

static uint8_t tmp421_read_t10(uint8_t msb_reg, uint8_t lsb_reg, int16_t *t10)
{
    uint8_t msb;
    uint8_t lsb;
    uint8_t frac;
    long value;

    if (!g_tmp421_addr) {
        *t10 = INVALID_T10;
        return 0;
    }

    if (!i2c_read_reg(g_tmp421_addr, msb_reg, &msb)) {
        *t10 = INVALID_T10;
        return 0;
    }
    if (!i2c_read_reg(g_tmp421_addr, lsb_reg, &lsb)) {
        lsb = 0;
    }
    if (msb_reg == TMP421_REG_REMOTE_MSB && (lsb & 0x03u)) {
        *t10 = INVALID_T10;
        return 0;
    }

    frac = (uint8_t)(lsb >> 4);
    value = ((long)msb * 160L + (long)frac * 10L + 8L) / 16L;
    value -= 640L;
    *t10 = (int16_t)value;
    return 1;
}

static uint16_t record_crc(const TempRecord *r)
{
    const uint16_t *w;
    uint8_t i;
    uint16_t crc;

    w = (const uint16_t *)r;
    crc = 0x5A5Au;
    for (i = 0; i < (uint8_t)((sizeof(TempRecord) / 2u) - 1u); i++) {
        crc = (uint16_t)((crc << 1) | (crc >> 15));
        crc ^= w[i];
    }
    return crc;
}

static uint8_t record_valid(const TempRecord *r)
{
    if (r->magic != RECORD_MAGIC) {
        return 0;
    }
    return (uint8_t)(record_crc(r) == r->crc);
}

static void flash_erase_segment(uint16_t addr)
{
    while (FCTL3 & BUSY) {
        ;
    }
    FCTL3 = FWKEY;
    FCTL1 = FWKEY | ERASE;
    *(volatile uint16_t *)addr = 0;
    while (FCTL3 & BUSY) {
        ;
    }
    FCTL1 = FWKEY;
    FCTL3 = FWKEY | LOCK;
}

static void flash_erase_log(void)
{
    uint16_t addr;

    __disable_interrupt();
    for (addr = FLASH_LOG_START; addr < FLASH_LOG_END; addr = (uint16_t)(addr + FLASH_SEGMENT_SIZE)) {
        flash_erase_segment(addr);
    }
    g_flash_next_index = 0;
    g_next_seq = 0;
    __enable_interrupt();
}

static void flash_write_record(uint16_t index, const TempRecord *r)
{
    const uint16_t *src;
    volatile uint16_t *dst;
    uint8_t words;
    uint8_t i;

    src = (const uint16_t *)r;
    dst = (volatile uint16_t *)(FLASH_LOG_START + index * sizeof(TempRecord));
    words = (uint8_t)(sizeof(TempRecord) / 2u);

    __disable_interrupt();
    while (FCTL3 & BUSY) {
        ;
    }
    FCTL3 = FWKEY;
    FCTL1 = FWKEY | WRT;
    for (i = 0; i < words; i++) {
        dst[i] = src[i];
    }
    while (FCTL3 & BUSY) {
        ;
    }
    FCTL1 = FWKEY;
    FCTL3 = FWKEY | LOCK;
    __enable_interrupt();
}

static void flash_scan(void)
{
    uint16_t i;
    const TempRecord *r;

    g_flash_next_index = 0;
    g_next_seq = 0;

    for (i = 0; i < FLASH_RECORD_COUNT; i++) {
        r = (const TempRecord *)(FLASH_LOG_START + i * sizeof(TempRecord));
        if (r->magic == 0xFFFFu) {
            g_flash_next_index = i;
            return;
        }
        if (!record_valid(r)) {
            g_flash_next_index = i;
            return;
        }
        g_flash_next_index = (uint16_t)(i + 1u);
        g_next_seq = (uint16_t)(r->seq + 1u);
    }
}

static void flash_log_sample(const TempSample *s)
{
    TempRecord r;

    if (g_flash_next_index >= FLASH_RECORD_COUNT) {
        flash_erase_log();
    }

    r.magic = RECORD_MAGIC;
    r.seq = g_next_seq++;
    r.die_t10 = s->die_t10;
    r.ntc_t10 = s->ntc_t10;
    r.tmp_local_t10 = s->tmp_local_t10;
    r.tmp_remote_t10 = s->tmp_remote_t10;
    r.flags = s->flags;
    r.crc = record_crc(&r);

    flash_write_record(g_flash_next_index, &r);
    g_flash_next_index++;
}

static uint16_t history_count(void)
{
    uint16_t i;
    const TempRecord *r;

    for (i = 0; i < FLASH_RECORD_COUNT; i++) {
        r = (const TempRecord *)(FLASH_LOG_START + i * sizeof(TempRecord));
        if (!record_valid(r)) {
            return i;
        }
    }
    return FLASH_RECORD_COUNT;
}

static uint8_t history_get(uint16_t index, TempRecord *out)
{
    const TempRecord *r;

    if (index >= FLASH_RECORD_COUNT) {
        return 0;
    }
    r = (const TempRecord *)(FLASH_LOG_START + index * sizeof(TempRecord));
    if (!record_valid(r)) {
        return 0;
    }
    *out = *r;
    return 1;
}

static void print_record_uart(const TempRecord *r)
{
    uart_putc('#');
    uart_print_u16(r->seq);
    uart_puts(" DIE=");
    uart_print_i16_t10(r->die_t10);
    uart_puts("C NTC=");
    uart_print_i16_t10(r->ntc_t10);
    uart_puts("C TMP=");
    uart_print_i16_t10(r->tmp_local_t10);
    uart_puts("C REM=");
    uart_print_i16_t10(r->tmp_remote_t10);
    uart_puts("C\r\n");
}

static void print_history_uart(void)
{
    uint16_t i;
    uint16_t count;
    TempRecord r;

    count = history_count();
    uart_puts("\r\nHistory count=");
    uart_print_u16(count);
    uart_puts("\r\n");

    for (i = 0; i < count; i++) {
        if (history_get(i, &r)) {
            print_record_uart(&r);
        }
    }
}

static void epd_bus_idle(void)
{
    EPD_CS_OUT |= EPD_CS_BIT;
    EPD_CLK_OUT &= ~EPD_CLK_BIT;
    EPD_SDI_OUT &= ~EPD_SDI_BIT;
    EPD_DC_OUT |= EPD_DC_BIT;
}

static void spi0_init(void)
{
    EPD_BUSY_SEL &= ~EPD_BUSY_BIT;
    EPD_RST_SEL &= ~EPD_RST_BIT;
    EPD_DC_SEL &= ~EPD_DC_BIT;
    EPD_CS_SEL &= ~EPD_CS_BIT;
    EPD_SDI_SEL &= ~EPD_SDI_BIT;
    EPD_CLK_SEL &= ~EPD_CLK_BIT;

    EPD_RST_OUT |= EPD_RST_BIT;
    epd_bus_idle();

    EPD_BUSY_DIR &= ~EPD_BUSY_BIT;
    EPD_RST_DIR |= EPD_RST_BIT;
    EPD_DC_DIR |= EPD_DC_BIT;
    EPD_CS_DIR |= EPD_CS_BIT;
    EPD_SDI_DIR |= EPD_SDI_BIT;
    EPD_CLK_DIR |= EPD_CLK_BIT;

    EPD_RST_OUT |= EPD_RST_BIT;
    epd_bus_idle();
}

static void epd_spi_delay(void)
{
    __delay_cycles(8);
}

static void spi0_write(uint8_t value)
{
    uint8_t bit;

    for (bit = 0; bit < 8u; bit++) {
        if (value & 0x80u) {
            EPD_SDI_OUT |= EPD_SDI_BIT;
        } else {
            EPD_SDI_OUT &= ~EPD_SDI_BIT;
        }
        epd_spi_delay();
        EPD_CLK_OUT |= EPD_CLK_BIT;
        epd_spi_delay();
        EPD_CLK_OUT &= ~EPD_CLK_BIT;
        value = (uint8_t)(value << 1);
        epd_spi_delay();
    }
}

static void epd_select(uint8_t selected)
{
    if (selected) {
        EPD_CS_OUT &= ~EPD_CS_BIT;
    } else {
        EPD_CS_OUT |= EPD_CS_BIT;
    }
}

static void epd_cmd(uint8_t cmd)
{
    epd_select(0);
    epd_select(1);
    EPD_CLK_OUT &= ~EPD_CLK_BIT;
    EPD_DC_OUT &= ~EPD_DC_BIT;
    spi0_write(cmd);
    epd_select(0);
}

static void epd_data(uint8_t data)
{
    epd_select(0);
    epd_select(1);
    EPD_CLK_OUT &= ~EPD_CLK_BIT;
    EPD_DC_OUT |= EPD_DC_BIT;
    spi0_write(data);
    epd_select(0);
}

static uint8_t epd_wait_busy(uint16_t timeout_ms)
{
    while ((EPD_BUSY_IN & EPD_BUSY_BIT) && timeout_ms > 0) {
        delay_ms(1);
        timeout_ms--;
    }
    if (EPD_BUSY_IN & EPD_BUSY_BIT) {
        return 0;
    }
    delay_ms(20);
    return 1;
}

static uint8_t epd_wait_update_done(void)
{
    uint16_t timeout_ms;
    uint8_t saw_busy;

    timeout_ms = EPD_BUSY_START_TIMEOUT_MS;
    saw_busy = 0;
    while (timeout_ms > 0) {
        if (EPD_BUSY_IN & EPD_BUSY_BIT) {
            saw_busy = 1;
            break;
        }
        delay_ms(1);
        timeout_ms--;
    }

    if (saw_busy) {
        if (!epd_wait_busy(EPD_BUSY_TIMEOUT_MS)) {
            return 0;
        }
    } else {
        delay_ms(EPD_POST_UPDATE_MS);
    }

    delay_ms(EPD_POST_UPDATE_MS);
    return saw_busy;
}

static void epd_reset(void)
{
    epd_bus_idle();
    EPD_RST_OUT |= EPD_RST_BIT;
    delay_ms(EPD_RESET_PRE_MS);
    EPD_RST_OUT &= ~EPD_RST_BIT;
    delay_ms(EPD_RESET_LOW_MS);
    EPD_RST_OUT |= EPD_RST_BIT;
    delay_ms(EPD_RESET_HIGH_MS);
    epd_bus_idle();
}

static uint8_t epd_soft_reset(void)
{
    uint16_t timeout_ms;
    uint8_t saw_busy;

    epd_cmd(0x12);
    delay_ms(1);

    timeout_ms = EPD_BUSY_START_TIMEOUT_MS;
    saw_busy = 0;
    while (timeout_ms > 0) {
        if (EPD_BUSY_IN & EPD_BUSY_BIT) {
            saw_busy = 1;
            break;
        }
        delay_ms(1);
        timeout_ms--;
    }

    if (saw_busy) {
        return epd_wait_busy(EPD_BUSY_TIMEOUT_MS);
    }

    delay_ms(100);
    return 1;
}

static void epd_recover_from_unknown_state(void)
{
    spi0_init();
    delay_ms(EPD_STARTUP_SETTLE_MS);
    epd_reset();
    (void)epd_wait_busy(EPD_BUSY_TIMEOUT_MS);
    (void)epd_soft_reset();
    (void)epd_wait_busy(EPD_BUSY_TIMEOUT_MS);
}

static uint8_t epd_alt_update(void)

    uint8_t saw_busy;

    epd_cmd(0x21);
    epd_data(0x83);
    epd_cmd(0x22);
    epd_data(0xC4);
    epd_cmd(0x20);
    delay_ms(100);
    saw_busy = epd_wait_update_done();

    epd_cmd(0x22);
    epd_data(0x03);
    epd_cmd(0x20);
    delay_ms(20);
    return saw_busy;
}

    epd_set_ram_cursor();

    delay_ms(5);

static void epd_pixel(uint16_t x, uint16_t y, uint8_t black)
{
    uint16_t ram_x;
    uint16_t ram_y;
    uint16_t index;
    uint8_t mask;

    if (x >= EPD_SCREEN_W || y >= EPD_SCREEN_H) {
        return;
    }

    ram_x = y;
    ram_y = x;
    index = (uint16_t)(ram_y * EPD_RAM_W_BYTES + (ram_x >> 3));
    mask = (uint8_t)(0x80u >> (ram_x & 7u));

    if (black) {
        epd_buf[index] &= (uint8_t)(~mask);
    } else {
        epd_buf[index] |= mask;
    }
}

static void epd_draw_char(uint16_t x, uint16_t y, char c, uint8_t scale)
{
    uint8_t col;
    uint8_t row;
    uint8_t sx;
    uint8_t sy;
    uint8_t bits;
    uint8_t glyph_index;

    if ((uint8_t)c < 32u || (uint8_t)c > 127u) {
        c = '?';
    }
    glyph_index = (uint8_t)c - 32u;

    for (col = 0; col < 5u; col++) {
        bits = font5x7[glyph_index][col];
        for (row = 0; row < 7u; row++) {
            if (bits & (1u << row)) {
                for (sx = 0; sx < scale; sx++) {
                    for (sy = 0; sy < scale; sy++) {
                        epd_pixel((uint16_t)(x + col * scale + sx),
                                  (uint16_t)(y + row * scale + sy), 1);
                    }
                }
            }
        }
    }
}

static void epd_draw_string(uint16_t x, uint16_t y, const char *s, uint8_t scale)
{
    uint16_t cursor;
    uint16_t step;

    cursor = x;
    step = (uint16_t)(6u * scale);
    while (*s && cursor < EPD_SCREEN_W) {
        epd_draw_char(cursor, y, *s++, scale);
        cursor = (uint16_t)(cursor + step);
    }
}

static uint8_t t10_fast_changed(int16_t now_t10, int16_t old_t10)
{
    int16_t diff;

    if (now_t10 == INVALID_T10 || old_t10 == INVALID_T10) {
        return 0;
    }

    diff = (int16_t)(now_t10 - old_t10);
    if (diff < 0) {
        diff = (int16_t)(-diff);
    }
    return (uint8_t)(diff >= EPD_FAST_REFRESH_DELTA_T10);
}

static uint8_t epd_current_refresh_due(const TempSample *s)
{
    uint8_t alarm_now;
    uint8_t alarm_last;

    if (!g_epd_has_last_sample) {
        return 1;
    }

    if (g_epd_samples_since_refresh >= EPD_AUTO_REFRESH_SAMPLES) {
        return 1;
    }

    alarm_now = sample_over_threshold(s);
    alarm_last = sample_over_threshold(&g_epd_last_sample);
    if (alarm_now && !alarm_last) {
        return 1;
    }

    if (t10_fast_changed(s->die_t10, g_epd_last_sample.die_t10)) {
        return 1;
    }
    if (t10_fast_changed(s->ntc_t10, g_epd_last_sample.ntc_t10)) {
        return 1;
    }
    if (t10_fast_changed(s->tmp_local_t10, g_epd_last_sample.tmp_local_t10)) {
        return 1;
    }
    if (t10_fast_changed(s->tmp_remote_t10, g_epd_last_sample.tmp_remote_t10)) {
        return 1;
    }

    return 0;
}

static void epd_note_current_refreshed(const TempSample *s)
{
    g_epd_last_sample = *s;
    g_epd_has_last_sample = 1;
    g_epd_samples_since_refresh = 0;
}

static void epd_force_next_current_refresh(void)
{
    g_epd_samples_since_refresh = EPD_AUTO_REFRESH_SAMPLES;
}

static void epd_show_current_auto(const TempSample *s)
{
    if (g_epd_samples_since_refresh < EPD_AUTO_REFRESH_SAMPLES) {
        g_epd_samples_since_refresh++;
    }

    if (!epd_current_refresh_due(s)) {
        return;
    }

    if (g_epd_driver == EPD_DRIVER_SPD2701) {
        epd_alt_show_current(s);
    } else {
        epd_show_current(s);
    }
    epd_note_current_refreshed(s);
}

        if (history_get((uint16_t)(start + row), &r)) {
            record_to_line(&r, line);
            epd_alt_draw_string(0, (uint16_t)(24u + row * 18u), line, 1);
        }
    }

    (void)epd_alt_write_buffer_to_screen(epd_buf);
}

    if (on && !g_buzzer_on) {
        BUZZER_OUT &= ~BUZZER_BIT;
        TA2CCR0 = (uint16_t)(BUZZER_HALF_PERIOD_TICKS - 1u);
        TA2CCTL0 = CCIE;
        TA2CTL = TASSEL_2 | MC_1 | TACLR;
        g_buzzer_on = 1;
    } else if (!on && g_buzzer_on) {
        TA2CTL = MC_0 | TACLR;
        TA2CCTL0 = 0;
        BUZZER_OUT &= ~BUZZER_BIT;
        g_buzzer_on = 0;
    }
}

static void sample_timer_init(void)
{
    TA0CCR0 = 32768u - 1u;
    TA0CCTL0 = CCIE;
    TA0CTL = TASSEL_1 | MC_1 | TACLR;
}

static void collect_sample(TempSample *s)
{
    uint16_t adc;
    uint8_t tmp_ok;
    uint8_t status;

    s->flags = 0;

    adc = adc_read_avg(ADC12INCH_10, ADC12SREF_1, 1);
    s->die_t10 = die_adc_to_t10(adc);
    s->flags |= FLAG_DIE_OK;

    adc = adc_read_avg(NTC_ADC_INCH, ADC12SREF_0, 0);
    g_last_ntc_adc = adc;
    if (adc > 4u && adc < 4091u) {
        s->ntc_t10 = ntc_adc_to_t10(adc);
        s->flags |= FLAG_NTC_OK;
    } else {
        s->ntc_t10 = INVALID_T10;
    }

    tmp_ok = tmp421_read_t10(TMP421_REG_LOCAL_MSB, TMP421_REG_LOCAL_LSB, &s->tmp_local_t10);
    if (tmp_ok) {
        s->flags |= FLAG_TMP_LOCAL_OK;
    }

    tmp_ok = tmp421_read_t10(TMP421_REG_REMOTE_MSB, TMP421_REG_REMOTE_LSB, &s->tmp_remote_t10);
    if (tmp_ok) {
        if (g_tmp421_addr && i2c_read_reg(g_tmp421_addr, TMP421_REG_STATUS, &status)) {
            if (status & BIT2) {
                s->tmp_remote_t10 = INVALID_T10;
            } else {
                s->flags |= FLAG_TMP_REMOTE_OK;
            }
        } else {
            s->flags |= FLAG_TMP_REMOTE_OK;
        }
    }
}

static void process_command(uint8_t cmd)
{
    uint16_t count;

    if (cmd >= 'A' && cmd <= 'Z') {
        cmd = (uint8_t)(cmd + ('a' - 'A'));
    }

    switch (cmd) {
    case 'h':
        print_history_uart();
        if (g_epd_driver == EPD_DRIVER_SPD2701) {
            epd_alt_show_history(0);
        } else {
            epd_show_history(0);
        }
        break;
    case 'n':
        count = history_count();
        if (count == 0) {
            if (g_epd_driver == EPD_DRIVER_SPD2701) {
                epd_alt_show_history(0);
            } else {
                epd_show_history(0);
            }
            break;
        }
        g_history_page = (uint16_t)(g_history_page + HISTORY_ROWS_ON_EPD);
        if (g_history_page >= count) {
            g_history_page = 0;
        }
        if (g_epd_driver == EPD_DRIVER_SPD2701) {
            epd_alt_show_history(g_history_page);
        } else {
            epd_show_history(g_history_page);
        }
        break;
    case 'c':
        epd_force_next_current_refresh();
        g_sample_due = 1;
        break;
    case '+':
        g_threshold_t10 = (int16_t)(g_threshold_t10 + 10);
        epd_force_next_current_refresh();
        g_sample_due = 1;
        uart_puts("Threshold=");
        uart_print_i16_t10(g_threshold_t10);
        uart_puts("C\r\n");
        break;
    case '-':
        g_threshold_t10 = (int16_t)(g_threshold_t10 - 10);
        epd_force_next_current_refresh();
        g_sample_due = 1;
        uart_puts("Threshold=");
        uart_print_i16_t10(g_threshold_t10);
        uart_puts("C\r\n");
        break;
    case 'e':
        flash_erase_log();
        uart_puts("History erased\r\n");
        if (g_epd_driver == EPD_DRIVER_SPD2701) {
            epd_alt_show_history(0);
        } else {
            epd_show_history(0);
        }
        break;
    case 'i':
        (void)tmp421_detect();
        i2c_print_scan();
        break;
    case 's':
        epd_print_pins();
        break;
    case 'x':
        epd_pin_drive_test();
        break;
    case 'p':
        g_epd_driver = EPD_DRIVER_SSD1673;
        epd_test_pattern();
        break;
    case 'o':
        epd_test_white();
        break;
    case 't':
        epd_test_text();
        break;
    case 'u':
        epd_test_bypass_cycle();
        break;
    case 'd':
        g_epd_driver = EPD_DRIVER_SPD2701;
        epd_alt_test_black();
        break;
    case 'f':
        epd_alt_test_white();
        break;
    case 'g':
        epd_alt_test_text();
        break;
    case 'm':
        g_epd_driver = EPD_DRIVER_SPD2701;
        g_epd_auto_update = 1;
        epd_alt_init();
        epd_force_next_current_refresh();
        g_sample_due = 1;
        uart_puts("EPD alternate driver selected\r\n");
        break;
    case 'r':
        g_epd_auto_update = 1;
        epd_force_next_current_refresh();
        g_sample_due = 1;
        uart_puts("EPD auto display resumed\r\n");
        break;
    case '?':
    default:
        uart_print_help();
        break;
    }
}

int main(void)
{
    TempSample sample;
    uint8_t cmd;

    WDTCTL = WDTPW | WDTHOLD;

    clock_init();
    gpio_init();
    uart_init();
    i2c_init();
    tmp421_init();
    flash_scan();
    epd_init();
    sample_timer_init();

    uart_puts("\r\nMSP430F5529 temperature logger\r\n");
    if (g_tmp421_addr) {
        uart_puts("TMP421 addr=");
        uart_print_hex_u8(g_tmp421_addr);
        uart_puts("\r\n");
    } else {
        uart_puts("TMP421 not detected; send 'i' for I2C scan\r\n");
    }
    uart_print_help();
    __enable_interrupt();

    while (1) {
        if (g_sample_due) {
            g_sample_due = 0;
            collect_sample(&sample);
            P1OUT ^= BIT0;
            uart_print_sample(&sample);
            flash_log_sample(&sample);
            buzzer_set(sample_over_threshold(&sample));
            if (g_epd_auto_update) {
                epd_show_current_auto(&sample);
            }
        }

        cmd = g_uart_cmd;
        if (cmd != 0) {
            g_uart_cmd = 0;
            process_command(cmd);
        }

        __bis_SR_register(LPM0_bits | GIE);
        __no_operation();
    }
}

#pragma vector=TIMER0_A0_VECTOR
__interrupt void TIMER0_A0_ISR(void)
{
    g_second_count++;
    if (g_second_count >= g_sample_interval) {
        g_second_count = 0;
        g_sample_due = 1;
        __bic_SR_register_on_exit(LPM0_bits);
    }
}

#pragma vector=TIMER2_A0_VECTOR
__interrupt void TIMER2_A0_ISR(void)
{
    BUZZER_OUT ^= BUZZER_BIT;
}

#pragma vector=USCI_A1_VECTOR
__interrupt void USCI_A1_ISR(void)
{
    uint8_t c;

    switch (__even_in_range(UCA1IV, 4)) {
    case 2:
        c = UCA1RXBUF;
        if (c != '\r' && c != '\n' && c != ' ' && c != '\t') {
            g_uart_cmd = c;
            __bic_SR_register_on_exit(LPM0_bits);
        }
        break;
    default:
        break;
    }
}
